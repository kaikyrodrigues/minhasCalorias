  <script>
    function calcularBiotipo() {
      let metabolismo = document.getElementById("metabolismo").value;
      let peso = document.getElementById("peso").value;
      let estrutura = document.getElementById("estrutura").value;
      let resultado = "";

      if (metabolismo === "rapido" && peso === "nao" && estrutura === "magro") {
        resultado = "Seu biotipo é: ECTOMORFO (magro, dificuldade para ganhar peso)";
      } 
      else if (metabolismo === "normal" && estrutura === "medio") {
        resultado = "Seu biotipo é: MESOMORFO (fácil ganho de massa muscular)";
      } 
      else if (metabolismo === "lento" && peso === "sim" && estrutura === "largo") {
        resultado = "Seu biotipo é: ENDOMORFO (tendência a acumular gordura)";
      } 
      else {
        resultado = "Você tem características mistas (biotipo misto).";
      }

      document.getElementById("resultado").innerText = resultado;
    }
  </script>
